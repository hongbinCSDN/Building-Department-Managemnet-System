--PSAC master table
create table P_MW_RECORD_PSAC(
    uuid VARCHAR2(36),
    record_id VARCHAR2(36),
    receive_date date,
    inspection_date date,
    CHK_MW_Not_Yet_Comm VARCHAR2(1),
    CHK_MW_Comm_And_Comp VARCHAR2(1),
    CHK_MW_In_Progress VARCHAR2(1),
    CHK_Inaccessible VARCHAR2(1),
    BS_NAME VARCHAR2(200),
    BS_POST VARCHAR2(200),
    BS_signature_date date,
    SO_NAME VARCHAR2(200),
    SO_POST VARCHAR2(200),
    SO_signature_date date,
    created_by VARCHAR2(20) not null,
    created_date date not null,
    modified_by VARCHAR2(20) not null,
    modified_date date not null,
    constraint P_MW_RECORD_PSAC_PK primary key (uuid),
    constraint P_MW_RECORD_PSAC_RECORD_ID_FK foreign key (record_id) references P_MW_RECORD(uuid)
);

--PSAC Image table
create table P_MW_RECORD_PSAC_Image(
    uuid VARCHAR2(36),
    P_MW_RECORD_PSAC_ID VARCHAR2(36),
    P_DSN_ID VARCHAR2(36),
    DSN_SUB VARCHAR2(50),
    FILE_PATH VARCHAR2(255),
    created_by VARCHAR2(20) not null,
    created_date date not null,
    modified_by VARCHAR2(20) not null,
    modified_date date not null,
    constraint P_MW_RECORD_PSAC_IMAGE_PK primary key (uuid),
    constraint P_MW_RECORD_PSAC_IMAGE_MASTER_FK foreign key (P_MW_RECORD_PSAC_ID) references P_MW_RECORD_PSAC(uuid),
    constraint P_MW_RECORD_PSAC_IMAGE_DSN_ID_FK foreign key (P_DSN_ID) references P_MW_DSN(uuid)
);
